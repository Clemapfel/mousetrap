## Script
cd mousetrap
mkdir build
cd build
cmake .. -DCMAKE_INSTALL_PREFIX=$prefix \
         -DMOUSETRAP_USE_GTK_OVERRIDES=ON \
         -DGTK4_INCLUDE_DIR=/workspace/destdir/include/gtk-4.0 \
         -DGLIB_INCLUDE_DIR="/workspace/destdir/include/glib-2.0;/workspace/destdir/include/glib-2.0/glib" \
         -DGLIB_LIB_DIR=/workspace/destdir/lib/glib-2.0/include \
         -DCAIRO_INCLUDE_DIR=/workspace/destdir/include/cairo \
         -DPANGO_INCLUDE_DIR=/workspace/destdir/include/pango-1.0/ \
	 -DHARFBUZZ_INCLUDE_DIR=/workspace/destdir/include/harfbuzz \
	 -DGDK_PIXBUF_INCLUDE_DIR=/workspace/destdir/include/gdk-pixbuf-2.0 \
	 -DGRAPHENE_INCLUDE_DIR=/workspace/destdir/include/graphene-1.0 \
	 -DGRAPHENE_LIB_DIR="/workspace/destdir/lib/graphene-1.0/include;/workspace/destdir/include/GL"
