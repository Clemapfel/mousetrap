## Script Linux
cd mousetrap
mkdir build
cd build
cmake .. -DCMAKE_INSTALL_LIBDIR=$prefix/lib64 \
         -DGTK4_INCLUDE_DIR=$prefix/include/gtk-4.0 \
         -DGLIB_INCLUDE_DIR="$prefix/include/glib-2.0;$prefix/include/glib-2.0/glib" \
         -DGLIB_LIB_DIR=$prefix/lib/glib-2.0/include \
         -DCAIRO_INCLUDE_DIR=$prefix/include/cairo \
         -DPANGO_INCLUDE_DIR=$prefix/include/pango-1.0/ \
         -DHARFBUZZ_INCLUDE_DIR=$prefix/include/harfbuzz \
         -DGDK_PIXBUF_INCLUDE_DIR=$prefix/include/gdk-pixbuf-2.0 \
         -DGRAPHENE_INCLUDE_DIR=$prefix/include/graphene-1.0 \
         -DGRAPHENE_LIB_DIR=$prefix/lib/graphene-1.0/include \
         -DGL_INCLUDE_DIR="$prefix/include/GL"

 make install -j 4

 # WINDOWS

 cmake .. -DCMAKE_INSTALL_PREFIX=$prefix -DCMAKE_INSTALL_LIBDIR=$prefix/lib64

 cd ../..

 cd build
 cmake .. -DCMAKE_INSTALL_PREFIX=$prefix \
     -DJulia_INCLUDE_DIRS="$prefix/include;$prefix/include/julia"
make install -j 4
